Router.post("/createproduct", upload.array("image"), async (req, res) => {
  const { error } = productValidation.validate(req.body);
  if (error) {
    return res.status(400).json({ message: error.details[0].message });
  }
  const uploader = async (file) =>
    await cloudinary.uploads(file, "../../images");
  try {
    if (req.body.image) {
      const imagedata = [];
      const images = req.body.image;
      console.log(req.body.amenities);
      console.log(images);

      for (const img of images) {
        const { file } = img;
        const newPath = await uploader(file);
        imagedata.push(newPath);
      }
      if (imagedata.length !== 0) {
        await productsModel.create({
          name: req.body.name,
          location: req.body.location,
          description: req.body.description,
          price: req.body.price,
          amenities: req.body.amenities,
          image: imagedata,
        });
        res.status(200).send({ message: "house created succesfully" });
      } else {
        res
          .status(400)
          .send({ message: "an error occurred during images upload" });
      }
    } else {
      res.status(400).send({ message: "an error occured" });
    }
  } catch (error) {
    res.status(500).send({ message: error.message });
  }
});
